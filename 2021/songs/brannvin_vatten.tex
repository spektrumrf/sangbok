\begin{song}{Brännvin, vatten}

	\original{\citetitle{jeepers_creepers}}
	
	Brännvin, vatten,\\
	smakar skit som katten.\\
	Brännvin helt rått,\\
	\begin{repetition}
		smakar jävligt gott.
	\end{repetition}
	
	Viinaa, vettä,\\
	mitä perkelettä?!\\
	Viinaa raakaa\\
	\begin{repetition}
		napaan kaatakaa!
	\end{repetition}
	
\end{song}
