\begin{song}{Sancta Lucia}
	
%	\original{Sancta Lucia}
	
	\original{\citetitle{sancta_lucia}}

    \showversenumber	
	Jag in silentio\\
	gör promenado\\
	nocturno solo\\
	in Esplanado.\\
	Jag där med grand favor\\
	väntar min pia\\
	Sancta Lucia, Sancta Lucia.
	
    \showversenumber
	Con vehementio\\
	slår jag rivale\\
	och etablerar\\
	grande scandale.\\
	Polisconstabelo\\
	otti in krago\\
	Sancta Lucia, Sancta Lucia.
	
    \showversenumber
	Jag skrek fortissimo:\\
	''Släpp mig för Phano!''\\
	Han röt crescendo:\\
	''Parla non guano!''\\
	Jag in arresto förs\\
	oh, infamia\\
	Sancta Lucia, Sancta Lucia.
	
    \showversenumber
	Oh, hårda destiné\\
	här in obscuro\\
	gråter jag lacrimae\\
	ödet är duro.\\
	Önskar al diavolo\\
	amore mia\\
	Satans Lucia, Satans Lucia.
	
\end{song}
