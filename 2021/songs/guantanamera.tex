\begin{song}{Guantanamera}
	
	\begin{repetition}
		Guantanamera,\\
		vi gör kondomer av lera,\\
		dom verkar inte fungera,\\
		för vi blir flera och flera.
	\end{repetition}
	
	Guantanamera,\\
	vi gör kondomer av \emph{gummi},\\
	dom verkar faktiskt fungera,\\
	för vi blir inte nå' flera.
	
\end{song}
