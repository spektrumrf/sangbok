\begin{song}{Moosen}
	
%	\original{Måsen/När månen vandrar på fästet blå}

	\original{\citetitle{nar_manen_vandrar} (\citetitle{skargardsidyll})}
	
	\addphrasetoindex{Det satt en älg}
	
	Det satt en älg i en klyvartopp,\\
	förklädd i älgjaktens månad.\\
	Den var befjädrad till horn och kropp,\\
	så skepparn blev rätt förvånad.\\
	''Jag är en mås, goa skepparn'', ljög den\\
	förklädda älgen, därefter flög den.\\
	Mjukt föll den se'n\\
	på skepparen.
	
\end{song}
