\begin{song}{Flottskaja}
	
%	\original{Bamsevisan}
	
	\original{\citetitle{svarta_ogon}}
	
	\addphrasetoindex{Äta flottskaja, dricka gottskaja}
	
	Äta flottskaja\\
	Dricka gottskaja\\
	Dricka merskaja\\
	Slå i flerskaja

	Å bli fullskaja\\
	Fall’ omkullskaja\\
	Om ja’ villskaja\\
	Ta en till.

\end{song}
