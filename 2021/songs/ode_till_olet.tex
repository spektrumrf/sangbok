\begin{song}{Ode till ölet}

    \addphrasetoindex{Tu-tu-tu-Tuborg}
    
    \original{\citetitle{trampa_pa_gasen}}
	
	\begin{repetition}
	Tu-tu-tu-Tuborg\\
	ca-ca-ca-Carlsberg\\
	är den bästa\\
	pi-pi-pi-pilsner som jag vet!
	\end{repetition}
	
	\begin{repetition}
	Tu-tu-tu-Carlsberg\\
	och ca-ca-ca-Tuborg\\
	är det bästa\\
	pi-pi-pi-ölet som jag vet!
	\end{repetition}
	
	\begin{repetition}
	Tu-tu-tu-ölberg\\
	och ca-ca-ca-pilsborg\\
	är den bästa\\
	pi-pi-pi-biran som jag vet!
	\end{repetition}
	
\end{song}
