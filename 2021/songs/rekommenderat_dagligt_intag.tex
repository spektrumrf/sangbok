\begin{song}{Rekommenderat dagligt intag}
	
%	\original{O Tannenbaum}
	
	\original{\citetitle{o_tannenbaum}}
	
	Min doktor sa\\
	att ''vin e' bra\\
	i dosen jämt\\
	ett glas per dag''.\\
	Men då jag inte drack igår\\
	och jag imorgon inget får\\
	så tänkte jag\\
	på medeltal\\
	då får jag trippeldos idag!
	
\end{song}
