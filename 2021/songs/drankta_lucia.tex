\begin{song}{Dränkta Lucia}
	
%	\original{Natten går tunga fjät}
	
	\original{\citetitle{sancta_lucia}}
	
	Huvet slår kopparslag,\\
	ögonen svider,\\
	magen i obehag,\\
	natten den lider.\\
	Då genom strupen går,\\
	hembränd en liten tår,\\
	vördat vare vårat brännvin,\\
	vörda vårt brännvin.
	
\end{song}
