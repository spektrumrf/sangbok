\begin{song}{Tidsmaskinen}
	
%	\original{Vem kan segla}\\
%	\songtext{Madeleine Ekblom, Lasse Lybeck, Jonas Westerlund}
	
	\original{\citetitle{vem_kan_segla}}\\
	\songtext{\citeauthor{ekblom}, \citeauthor{lybeck} och \citeauthor{westerlund}}
	
	\addphrasetoindex{Vem kan bygga en tidsmaskin}
	
	Vem kan bygga en tidsmaskin?\\
	Vem kan få ett nobelpris?\\
	Vem kan koda med vännen sin,\\
	utan att börja gräla?
	
	Jag kan bygga en tidsmaskin!\\
	Jag kan få ett nobelpris!\\
	Men ej koda med vännen min,\\
	utan att börja gräla.
	
\end{song}
