\begin{song}{Baklängessupen}
	
%	\original{Hej tomtegubbar}
	
	\original{\citetitle{hej_tomtegubbar}}
	
	\addphrasetoindex{Toj, hämtigubbar, gla i såsen}
	
	\begin{repetition}
		Toj, hämtigubbar, gla i såsen\\
		och kvasi vastiga lura.
	\end{repetition}\\
	Ten hiten lid vi heva lär\\
	med möcken myda och svärt bestor.\\
	Toj, hämtigubbar, gla i såsen\\
	och kvasi vastiga lura!
	
\end{song}
