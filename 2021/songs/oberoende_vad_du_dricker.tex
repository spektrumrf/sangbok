\begin{song}{Oberoende vad du dricker}

	\original{\citetitle{never_gonna_give_you_up}}\\
	\songtext{\citeauthor{vuorela}}

    \addphrasetoindex{Alltid kan man ta en till}
	
	Allti kan man ta en till,\\
	eller bara säga nej\\
	de e ändå upp till dig,\\
	de e ditt val!

	När du ändå kommit hit\\
	ska du ha en bra kväll,\\
	oberoende vad du dricker,\\
	De blir bra!

	
\end{song}
