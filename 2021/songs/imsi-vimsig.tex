\begin{song}{Imsi-vimsig}
	
%	\original{Imse-vimse spindel}
	
	\original{\citetitle{imse-vimse_spindel}}
	
	Imsi-vimsig blir man\\
	av en liten hutt.\\
	Pulsen börjar öka,\\
	hjärtat tar ett skutt.\\
	Knäna skälver,\\			%<- fattas det två stavelser?!? % Nä!
	näsan den blir blå.\\
	Fast det är så läskigt\\
	vågar vi ändå.
	
	Imsig vimsig blir man\\
	utav lite vin.\\
	Klättrar uppå stolen,\\
	verkar piggelin.\\
	Ramlar under bordet,\\
	sussar en minut.\\
	Vaknar av att vinet\\
	i glaset har tatt slut.
	
\end{song}
