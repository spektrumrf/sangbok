\begin{song}{Sup i dag}
	
%	\original{Vinden drar}
	
	\original{\citetitle{vinden_drar}}
	
	Sup i dag, sup i går,\\
	sup i övermorgon.\\
	Supa lite varje dag,\\
	det fördelar sorgen.
	
	Full i dag full i går,\\
	full i övermorgon.\\
	Liten fylla varje dag,\\
	det fördelar sorgen.
	
	Puss på mun, puss på kinn,\\
	puss på alla ställen.\\
	Pussa lite här och där,\\
	det förgyller kvällen.
	
\end{song}
