\begin{song}{Spektrumvisan}
	
%	\original{När som sädesfälten}\\
%	\songtext{Torsten Nilsson}
	
	\original{\citetitle{barndomshemmet}}\\
	\songtext{\citeauthor{nilsson}}
	
	Liksom solens strålar\\
	bryts uti kristaller,\\
	liksom färg som blandas ut\\
	på en palett,\\
	lyser SPEKTRUM fram ur\\
	välfyllda pokaler.\\
	Det är det skönaste\\
	vi någonsin har sett!
	
	Man förälskar sig i\\
	Aurans aurora,\\
	och man dricker därför\\
	glas på glas på glas.\\
	Man blir helt förtrollad\\
	utav koskens gloria,\\
	javisst fasen har man\\
	kul på vårt kalas!
	
\end{song}
